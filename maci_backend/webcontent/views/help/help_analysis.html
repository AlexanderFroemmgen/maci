Tweak your Figures

Provide a modified callback function to tweak generated figures. In the following, we collected a few snippets we found useful.

def my_callback():
    # set x label
    plt.xlabel("")

    # set y label
    plt.ylabel("Size [Bytes]")

    # set y range
    plt.ylim([0, 4000000])

    # use log scale
    plt.yscale('log')

    # set plot title
    plt.title("Total Message Overhead")
    
    # set plot size
    fig = plt.gcf()
    fig.set_size_inches(4, 3) # width and height

    # set x tick labels
    ax = plt.gca()
    ax.set_xticklabels(["foo", "bar"], rotation=30)

    # add an additional line
    # x1, x2, y1, y2
    plt.plot([0.0, 10000000], [500, 500], 'k--', lw=1, label="Storage boundary")

    # scale values (y-axis)
    # import matplotlib.ticker as ticker
    scale_y = 1000.0
    ticks_y = ticker.FuncFormatter(lambda y, pos: '{0:g}'.format(y/scale_y))
    ax.yaxis.set_major_formatter(ticks_y)
